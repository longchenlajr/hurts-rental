------------------------------------------------------------------------------------

CREATE TABLE dropoff(
	dropoff_id INTEGER GENERATED ALWAYS as IDENTITY,
	charge INTEGER,
	PRIMARY KEY(dropoff_id)
);

CREATE TABLE fuel(
	fuel_id INTEGER GENERATED ALWAYS as IDENTITY,
	remaining_fuel INTEGER NOT NULL,
	PRIMARY KEY(fuel_id)
);

CREATE TABLE insurance(
	insurance_id INTEGER GENERATED ALWAYS as IDENTITY,
	rate INTEGER NOT NULL,
	type VARCHAR(16) NOT NULL, 
	periodical VARCHAR(16) NOT NULL,
	PRIMARY KEY(insurance_id)
);

CREATE TABLE misc(
	miscellaneous_id INTEGER GENERATED ALWAYS as IDENTITY,
	child_seat NUMBER(1) DEFAULT 0 NOT NULL,
	nav NUMBER(1) DEFAULT 0 NOT NULL,
	radio NUMBER(1) DEFAULT 0 NOT NULL,
	PRIMARY KEY(miscellaneous_id)
);


CREATE TABLE discount(
	discount_id INTEGER GENERATED ALWAYS as IDENTITY,
	percent_discount INTEGER NOT NULL,
	PRIMARY KEY(discount_id)
);

CREATE TABLE membership(
	membership_id INTEGER GENERATED ALWAYS as IDENTITY,
	m_num INTEGER NOT NULL,
	m_name VARCHAR(64) NOT NULL,
	discount_id INTEGER NOT NULL,
	PRIMARY KEY(membership_id),
	FOREIGN KEY(discount_id) REFERENCES discount(discount_id)
);

CREATE TABLE location(
	location_id INTEGER GENERATED ALWAYS as IDENTITY,
	address VARCHAR(64) NOT NULL,
	city VARCHAR(64) NOT NULL,
	state VARCHAR(64) NOT NULL,
	PRIMARY KEY(location_id)
);

CREATE TABLE vehicle(
	vehicle_id INTEGER GENERATED ALWAYS as IDENTITY,
	make VARCHAR(64) NOT NULL,
	model VARCHAR(64) NOT NULL,
	plate VARCHAR(16) NOT NULL,
	type VARCHAR(32) NOT NULL,
	odometer INTEGER NOT NULL,
	reserved NUMBER(1) DEFAULT 0 NOT NULL,
	instock NUMBER(1) DEFAULT 0 NOT NULL,
	fuel_id INTEGER NOT NULL,
	location_id INTEGER NOT NULL,
	vin VARCHAR(32) NOT NULL, 
	PRIMARY KEY(vehicle_id),
	FOREIGN KEY(fuel_id) REFERENCES fuel(fuel_id),
	FOREIGN KEY(location_id) REFERENCES location(location_id)
);

CREATE TABLE customer(
	customer_id INTEGER GENERATED ALWAYS as IDENTITY,
	f_name VARCHAR(32) NOT NULL,
	l_name VARCHAR(32) NOT NULL,
	address VARCHAR(64) NOT NULL,
	DLN INTEGER NOT NULL,
	membership_id INTEGER NOT NULL,
	PRIMARY KEY(customer_id),
	FOREIGN KEY(membership_id) REFERENCES membership(membership_id)
);


CREATE TABLE rental(
	rental_id INTEGER GENERATED ALWAYS as IDENTITY,
	start_date DATE NOT NULL,
	end_date DATE NOT NULL,
	base_rate INTEGER NOT NULL,
	customer_id INTEGER NOT NULL,
	vehicle_id INTEGER NOT NULL,
	PRIMARY KEY(rental_id),
	FOREIGN KEY(customer_id) REFERENCES customer(customer_id),
	FOREIGN KEY(vehicle_id) REFERENCES vehicle(vehicle_id)
);

CREATE TABLE payment(
	payment_id INTEGER GENERATED ALWAYS as IDENTITY,
	type VARCHAR(16) NOT NULL,
	card_number INTEGER NOT NULL,
	cvv INTEGER NOT NULL,
	customer_id INTEGER NOT NULL,
	PRIMARY KEY(payment_id),
	FOREIGN KEY(customer_id) REFERENCES customer(customer_id)
);

CREATE TABLE charges(
	charges_id INTEGER GENERATED ALWAYS as IDENTITY,
	charges INTEGER NOT NULL,
	miscellaneous_id INTEGER,
	insurance_id INTEGER,
	dropoff_id INTEGER,
	fuel_id INTEGER,
	payment_id INTEGER NOT NULL,
	PRIMARY KEY(charges_id),
	FOREIGN KEY(miscellaneous_id) REFERENCES misc(miscellaneous_id),
	FOREIGN KEY(insurance_id) REFERENCES insurance(insurance_id),
	FOREIGN KEY(dropoff_id) REFERENCES dropoff(dropoff_id),
	FOREIGN KEY(fuel_id) REFERENCES fuel(fuel_id), 
	FOREIGN KEY(payment_id) REFERENCES payment(payment_id)
);
